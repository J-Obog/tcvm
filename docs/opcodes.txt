<alpha> ::= 'a'->'z' | 'A'->'Z'
<digit> ::= '0'->'9'
<number> ::= <digit> { <digit> }
<identifier> ::= <alpha> { <alpha> | <digit> } 
<condition> :: = 'NZ' | 'Z' | 'NC' | 'C' | 'MP' 
<type_specifier> ::= 'B' | 'W' | 'DW'

REGISTERS
<gp_register_range> ::= '0' | '1' | '2' | '3' | '4' | '5' | '6' | '7'
<gp_register> ::= '$' 'r' <gp_register_range>
<sp_register> ::= '$' 'sp' | 'rar'
<register> ::= <gp_register> | <sp_register>


ADDRESS MODE
<source> ::= <identifier> | <immediate> | <register>
<destination> ::= '[' <register> ']' | <gp_register>
<destination_source_pair> ::= ' ' <destination> ',' <source> 


INSTRUCTION SET
<label_directive> ::= 'LABEL' <identifier>
<data_directive> ::= 'DATA' <type_specifier> <immediate>
<move_instruction> ::= 'MOV' <type_specifier> <destination_source_pair>
<arithmetic_instruction> ::= ('ADD' | 'SUB' | 'MUL' | 'DIV') <destination_source_pair>
<logical_instruction> ::= ('AND' | 'OR' | 'XOR' | 'NOT') <destination_source_pair>
<compare_instruction> ::= 'CMP' <destination_source_pair>
